// ── Combination helper (for coverage chart) ───────────────────────────────────
function combination($n, $k)
{
    if ($k > $n) { return 0; }
    $res = 1;
    for ($i = 1; $i <= $k; $i++) {
        $res = $res * ($n - $i + 1) / $i;
    }
    return round($res);
}

    return is_int($n) && $n >= $pickSize && $n <= $mainNumbersMax;
})));
sort($options);

$stats = [];
foreach ($options as $opt) {
    $ways = combination($opt, $pickSize);
    $odds = $ways ? round($totalCombinations / $ways) : 0;
    $stats[] = [
        'count' => $opt,
        'ways'  => $ways,
        'odds'  => $odds,
        'pct'   => $ways ? round(100 * $ways / $totalCombinations, 2) : 0,
    ];
}

function leFmtDate(?string $date): string
{

    $ts = strtotime($date);

}

function leFmtDateLong(?string $date): string
{

}

function lePad2(string $value): string
{
    $value = trim((string) $value);
    if ($value === '') { return ''; }
    if (ctype_digit($value) && (int) $value < 10) {
        return str_pad($value, 2, '0', STR_PAD_LEFT);
    }
    return $value;
}

function leFindRepeatedFromLatest(array $latestBalls, array $previousRows): array
{
    $repeated = [];
    foreach ($previousRows as $row) {
        $prevBalls = [
            trim((string) ($row['first']  ?? '')),
            trim((string) ($row['second'] ?? '')),
            trim((string) ($row['third']  ?? '')),
            trim((string) ($row['fourth'] ?? '')),
            trim((string) ($row['fifth']  ?? '')),
        ];
        foreach ($latestBalls as $ball) {
            if ($ball !== '' && in_array($ball, $prevBalls, true) && !in_array($ball, $repeated, true)) {
                $repeated[] = $ball;
            }
        }
    }
    sort($repeated, SORT_NATURAL);
    return $repeated;
}

/**
 * Previous occurrence date for a ball.
 * For EuroMillions bonus (Lucky Stars): checks BOTH sixth and seventh columns.
 */
function leGetPreviousOccurrenceDate(
    \Joomla\Database\DatabaseDriver $db,
    string $dbCol,
    string $gameId,
    string $drawDate,
    string $ball,
    bool $isBonus = false
): ?string {
    if ($ball === '') { return null; }
    $query = $db->getQuery(true)
        ->select('MAX(' . $db->quoteName('draw_date') . ')')
        ->from($db->quoteName($dbCol))
        ->where($db->quoteName('game_id') . ' = ' . $db->quote($gameId))
        ->where($db->quoteName('draw_date') . ' < ' . $db->quote($drawDate));
    if ($isBonus) {
        // EuroMillions: Lucky Star can appear in sixth OR seventh
        $query->where(
            '(' .
            $db->quoteName('sixth')   . ' = ' . $db->quote($ball) . ' OR ' .
            $db->quoteName('seventh') . ' = ' . $db->quote($ball) .
            ')'
        );
    } else {
        $query->where(
            '(' .
            $db->quoteName('first')  . ' = ' . $db->quote($ball) . ' OR ' .
            $db->quoteName('second') . ' = ' . $db->quote($ball) . ' OR ' .
            $db->quoteName('third')  . ' = ' . $db->quote($ball) . ' OR ' .
            $db->quoteName('fourth') . ' = ' . $db->quote($ball) . ' OR ' .
            $db->quoteName('fifth')  . ' = ' . $db->quote($ball) .
            ')'
        );
    }
    $db->setQuery($query);
    $result = $db->loadResult();
    return $result ? (string) $result : null;
}

/**
 * Normalize play — max=50 for EuroMillions 1–50 main range.
 */
function leNormalizePlay(array $numbers, int $min = 1, int $max = 50, int $limit = 5): array
{
    $ints = [];
    $seen = [];
    foreach ($numbers as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n < $min || $n > $max || isset($seen[$n])) { continue; }
        $seen[$n] = true;
        $ints[] = $n;
        if (count($ints) >= $limit) { break; }
    }
    sort($ints, SORT_NUMERIC);
    $out = [];
    foreach ($ints as $n) {
        $out[] = ($n < 10) ? '0' . $n : (string) $n;
    }
    return $out;
}

/**
 * Build play from pools — max=50 for EuroMillions 1–50 main range.
 */
function leBuildPlayFromPools(array $pools, array $fallbackPool, int $limit = 5): array
{
    $picked = [];
    $seen   = [];
    foreach ($pools as $pool) {
        foreach ($pool as $value) {
            $n = is_int($value) ? $value : leNumberToInt((string) $value);
            if ($n < 1 || $n > 50 || isset($seen[$n])) { continue; } // max 50 for EuroMillions
            $seen[$n] = true;
            $picked[] = $n;
            if (count($picked) >= $limit) { break 2; }
        }
    }
    if (count($picked) < $limit) {
        foreach ($fallbackPool as $value) {
            $n = is_int($value) ? $value : leNumberToInt((string) $value);
            if ($n < 1 || $n > 50 || isset($seen[$n])) { continue; }
            $seen[$n] = true;
            $picked[] = $n;
            if (count($picked) >= $limit) { break; }
        }
    }
    return leNormalizePlay($picked, 1, 50, $limit);
}

function leConsecutiveCount(array $numbers): int
{
    $sorted = [];
    foreach ($numbers as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n > 0) { $sorted[] = $n; }
    }
    sort($sorted, SORT_NUMERIC);
    $count = 0;
    for ($i = 1; $i < count($sorted); $i++) {
        if (($sorted[$i] - $sorted[$i - 1]) === 1) { $count++; }
    }
    return $count;
}

function leArithmeticPatternScore(array $numbers): int
{
    $sorted = [];
    foreach ($numbers as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n > 0) { $sorted[] = $n; }
    }
    sort($sorted, SORT_NUMERIC);
    if (count($sorted) < 4) { return 0; }
    $sameDiff = 0;
    for ($i = 2; $i < count($sorted); $i++) {
        $d1 = $sorted[$i - 1] - $sorted[$i - 2];
        $d2 = $sorted[$i] - $sorted[$i - 1];
        if ($d1 === $d2) { $sameDiff++; }
    }
    return $sameDiff;
}

/**
 * Combination profile — thresholds updated for EuroMillions 5/50 matrix.
 * Low threshold: ≤25 (half of 50). Density: tight ≤18, moderate ≤30.
 */
function leBuildCombinationProfile(array $play, array $latestMain): array
{
    $ints = [];
    foreach ($play as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n > 0) { $ints[] = $n; }
    }
    sort($ints, SORT_NUMERIC);

    $odd = 0;
    $low = 0;
    foreach ($ints as $n) {
        if (($n % 2) !== 0) { $odd++; }
        if ($n <= 25) { $low++; } // low threshold for 1–50 range
    }

    $sum         = array_sum($ints);
    $spread      = (!empty($ints)) ? (max($ints) - min($ints)) : 0;
    $consecutive = leConsecutiveCount($ints);
    $repeated    = 0;

    $latestLookup = [];
    foreach ($latestMain as $value) {
        $latestLookup[lePad2((string) $value)] = true;
    }
    foreach ($ints as $n) {
        $key = ($n < 10) ? '0' . $n : (string) $n;
        if (isset($latestLookup[$key])) { $repeated++; }
    }

    // Density thresholds proportional to 1–50 range
    $density = 'Wide spread';
    if ($spread <= 18) {
        $density = 'High cluster density';
    } elseif ($spread <= 30) {
        $density = 'Moderate cluster density';
    }

    return [
        'odd_even'        => $odd . '/' . (count($ints) - $odd),
        'low_high'        => $low . '/' . (count($ints) - $low),
        'sum'             => $sum,
        'spread'          => $spread,
        'consecutive'     => $consecutive,
        'repeated'        => $repeated,
        'cluster_density' => $density,
    ];
}

/**
 * Crowd risk profile — spread thresholds updated for 1–50 range.
 * Birthday numbers (≤31) threshold unchanged — same principle applies.
 */
function leCrowdRiskProfile(array $play, array $favoriteNumbers): array
{
    $ints = [];
    foreach ($play as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n > 0) { $ints[] = $n; }
    }
    sort($ints, SORT_NUMERIC);

    $under31        = 0;
    $favHits        = 0;
    $favoriteLookup = [];
    foreach ($favoriteNumbers as $fav) { $favoriteLookup[(int) $fav] = true; }

    foreach ($ints as $n) {
        if ($n <= 31) { $under31++; }
        if (isset($favoriteLookup[$n])) { $favHits++; }
    }

    $spread       = (!empty($ints)) ? (max($ints) - min($ints)) : 0;
    $consecutive  = leConsecutiveCount($ints);
    $visualPattern = leArithmeticPatternScore($ints) >= 2;

    $score   = 0;
    $reasons = [];

    if ($under31 >= 4) {
        $score += 2;
        $reasons[] = 'birthday-heavy composition';
    } elseif ($under31 >= 3) {
        $score += 1;
        $reasons[] = 'leaning toward under-31 numbers';
    }

    if ($consecutive >= 2) {
        $score += 2;
        $reasons[] = 'sequence concentration';
    } elseif ($consecutive === 1) {
        $score += 1;
        $reasons[] = 'minor sequence pattern';
    }

    if ($visualPattern) {
        $score += 1;
        $reasons[] = 'obvious spacing pattern';
    }

    // Spread thresholds scaled for 1–50 range
    if ($spread <= 18) {
        $score += 2;
        $reasons[] = 'tight clustering';
    } elseif ($spread <= 28) {
        $score += 1;
        $reasons[] = 'moderate clustering';
    }

    if ($favHits >= 2) {
        $score += 1;
        $reasons[] = 'cultural favorite overlap';
    }

    $label = 'Low crowd risk';
    if ($score >= 5)      { $label = 'High crowd risk'; }
    elseif ($score >= 3)  { $label = 'Medium crowd risk'; }

    $explanation = !empty($reasons)
        ? ('Signals: ' . implode(', ', $reasons) . '.')
        : 'Signals: broad spread and lower overlap with common crowd patterns.';

    return [
        'score'       => $score,
        'label'       => $label,
        'explanation' => $explanation,
        'under31'     => $under31,
    ];
}

// Helper: pick 2 distinct Lucky Stars from a primary + secondary source
function lePickTwoStars(array $primary, array $secondary, string $default1 = '01', string $default2 = '02'): array
{
    $s1 = !empty($primary) ? (string) $primary[0] : $default1;
    // find s2 different from s1
    $s2 = $default2;
    foreach (array_merge($secondary, $primary) as $candidate) {
        if ((string) $candidate !== $s1) { $s2 = (string) $candidate; break; }
    }
    if ($s2 === $s1) { $s2 = ($s1 === $default1) ? $default2 : $default1; }
    return [$s1, $s2];
}

  function buildModel(inputShape, settings) {
    var input = tf.input({ shape: [inputShape] });
    var x = tf.layers.dense({ units: 128, activation: settings.ACTIVATION_FUNCTION }).apply(input);
    x = tf.layers.dropout({ rate: settings.DROPOUT_RATE }).apply(x);
    for (var i = 0; i < settings.HIDDEN_LAYERS; i++) {
      x = tf.layers.dense({ units: 64, activation: settings.ACTIVATION_FUNCTION }).apply(x);
      x = tf.layers.dropout({ rate: settings.DROPOUT_RATE }).apply(x);
    }
    var mainNumbersOutput = tf.layers.dense({ units: mainNumbersMax, activation: 'sigmoid', name: 'main_numbers_output' }).apply(x);
    var outputs, model;
    if (hasExtraBall) {
      var extraOutput = tf.layers.dense({ units: extraBallMax, activation: 'sigmoid', name: 'extra_output' }).apply(x);
      outputs = [mainNumbersOutput, extraOutput];
    } else {
      outputs = mainNumbersOutput;
    }
    model = tf.model({ inputs: input, outputs: outputs });
    var optimizer = tf.train.adam(settings.LEARNING_RATE);
    model.compile({
      optimizer: optimizer,
      loss: hasExtraBall ? ['binaryCrossentropy', 'binaryCrossentropy'] : 'binaryCrossentropy',
      metrics: ['accuracy']
    });
    return model;
  }

        breakdownHTML += '[[tr]][[td]]' + item.number + '[[/td]][[td]]' + (item.probability * 100).toFixed(2) + '%[[/td]][[/tr]]';
      });
      breakdownHTML += '[[/tbody]][[/table]]';
      if (hasExtraBall && extraPrediction) {
        var extraProbs = Array.from(extraPrediction).map(function(prob, idx) { return { number: idx + 1, probability: prob }; });
        extraProbs.sort(function(a, b) { return b.probability - a.probability; });
        breakdownHTML += '[[h3]]Top Lucky Star Suggestions (from 1&ndash;12 pool)[[/h3]][[table]][[thead]][[tr]][[th]]Lucky Star[[/th]][[th]]Probability (%)[[/th]][[/tr]][[/thead]][[tbody]]';
        extraProbs.slice(0, 3).forEach(function(item) {
          breakdownHTML += '[[tr]][[td]]' + item.number + '[[/td]][[td]]' + (item.probability * 100).toFixed(2) + '%[[/td]][[/tr]]';
        });
        breakdownHTML += '[[/tbody]][[/table]]';
      }
      if (numberBreakdownTableDiv) numberBreakdownTableDiv.innerHTML = breakdownHTML;
      if (progressMessage) progressMessage.innerText = messages[8];
      var predictionDate = new Date();
      var predictionDateString = predictionDate.toLocaleString();
      if (predictionDateValue) predictionDateValue.innerText = predictionDateString;
      var lotteryNameJS = <?php echo json_encode($lotteryName); ?>;
      var saveGenAt = document.getElementById('saveGeneratedAt');
      var saveLbl   = document.getElementById('saveLabel');
      if (saveGenAt) saveGenAt.value = predictionDateString;
      if (saveLbl)   saveLbl.value   = lotteryNameJS + ' Prediction \u2013 ' + predictionDateString;
      var saveEpochs  = document.getElementById('saveEpochs');
      var saveBatch   = document.getElementById('saveBatchSize');
      var saveDrop    = document.getElementById('saveDropoutRate');
      var saveLR      = document.getElementById('saveLearningRate');
      var saveAF      = document.getElementById('saveActivationFunction');
      var saveHL      = document.getElementById('saveHiddenLayers');
      if (saveEpochs) saveEpochs.value = EPOCHS;
      if (saveBatch)  saveBatch.value  = BATCH_SIZE;
      if (saveDrop)   saveDrop.value   = DROPOUT_RATE;
      if (saveLR)     saveLR.value     = LEARNING_RATE;
      if (saveAF)     saveAF.value     = ACTIVATION_FUNCTION;
      if (saveHL)     saveHL.value     = HIDDEN_LAYERS;
      var mainPredFull = mainProbs.map(function(item) { return item.number; }).join(',');
      var extraPredFull = '';
      if (hasExtraBall && extraPrediction) {
        var ep2 = Array.from(extraPrediction).map(function(prob, idx) { return { number: idx + 1, probability: prob }; });
        ep2.sort(function(a, b) { return b.probability - a.probability; });
        extraPredFull = ep2.map(function(item) { return item.number; }).join(',');
      }
      var saveMain = document.getElementById('saveMainNumbers');
      var saveExt  = document.getElementById('saveExtraBallNumbers');
      if (saveMain) saveMain.value = mainPredFull;
      if (saveExt)  saveExt.value  = extraPredFull;
      if (enableBacktestCheckbox && enableBacktestCheckbox.checked) {
        if (backtestNoticeDiv) backtestNoticeDiv.style.display = 'block';
        if (backtestProgressContainer) backtestProgressContainer.style.display = 'block';
        if (countdownDiv) countdownDiv.innerHTML = 'Running backtesting on previous draws...';
        var sortedDrawsAsc = filteredDraws.slice().sort(function(a, b) { return new Date(a.draw_date) - new Date(b.draw_date); });
        var backtestResults = await runBacktesting(sortedDrawsAsc, settings);
        var backtestHTML = renderBacktestResults(backtestResults);
        if (backtestResultsDiv) { backtestResultsDiv.innerHTML = backtestHTML; backtestResultsDiv.style.display = 'block'; }
        if (backtestNoticeDiv) backtestNoticeDiv.innerHTML = '[[h4]]Backtesting Complete[[/h4]][[p]]Backtesting has finished processing all draws.[[/p]]';
        if (backtestProgressContainer) backtestProgressContainer.style.display = 'none';
        if (countdownDiv) countdownDiv.innerHTML = 'Done! Analysis is complete.';
      }
      setTimeout(function() {
        if (loadingDiv) loadingDiv.style.display = 'none';
        if (predictionDiv) predictionDiv.style.display = 'block';
        if (numberBreakdownDiv) numberBreakdownDiv.style.display = 'block';
      }, 1000);
      clearInterval(funMessageTimer);
    } catch (error) {
      console.error('Error during AI analysis:', error);
      if (loadingDiv) loadingDiv.innerText = 'An error occurred during analysis: ' + error.message;
      startButton.style.display = 'block';
    }
  });
